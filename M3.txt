    return f"It is 30°C and sunny in {city}"

def ask_claude(user_question):
    # Very simple logic to simulate Claude thinking
    if "weather" in user_question.lower() or "umbrella" in user_question.lower():
        
        # Extract city (simple assumption: last word is city)
        city = user_question.split()[-1].replace("?", "")
        
        # Call our tool
        weather = get_weather(city)
        
        return f"{weather}. You can plan accordingly!"
    
    else:
        return "This question is not related to weather."
        


print(ask_claude("Should I carry an umbrella in Jaipur today?"))


print(ask_claude("What is the capital of Rajasthan?"))
